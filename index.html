Euclidean Rhythm - 5/9


var Beet = require('beet.js');
var context = new AudioContext();
var beet = new Beet({
  context: context
});
var pattern = beet.pattern(5, 9); // returns 101110111
var layer = beet.layer(pattern, callback);
beet.add(layer).start();
